# Format: ENTITY_NAME: regex_pattern

# VARCODE is a fictional identifier that won't be confused with real entities
# Match patterns like TEST_VARCODE1, PROD_VARCODE99, DEV_VARCODE123 (prefix with underscore)
VARCODE: \b(?:TEST_|PROD_|DEV_)VARCODE\d+\b

# Match VARCODE followed by alphanumeric (no prefix/suffix allowed)
VARCODE: \bVARCODE[a-z0-9]{1,6}\b

# Match VARCODE with 1-6 digits exactly
VARCODE: \bVARCODE\d{1,6}\b
